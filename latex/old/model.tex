\section{Methods}
\subsection{Discrete-Time SIR Dynamics}

The compartments evolve according to the discrete-time SIR model:
\begin{align}\begin{split}%\label{eq:sir}
S_i(t+1) &= S_i(t) - \beta(t) S_i(t) \sum_j C_{ij} \frac{I_j(t)}{N_j} \\
I_i(t+1) &= I_i(t) + \beta(t) S_i(t) \sum_j C_{ij} I_j(t)\frac{I_j(t)}{N_j} - \gamma I_i(t) \\
R_i(t+1) &= R_i(t) + \gamma I_i(t)
\end{split}\end{align}

We consider two regions with connectivity matrix:
\begin{equation}
\mathbf{C} = \begin{bmatrix} 1-\theta & \theta \\ \theta & 1-\theta \end{bmatrix}
\end{equation}

If we define the incidence at time $t$ as $\mu(t) = \beta(t) S(t)
\circ CI(t)$ then these equations simply become:
\begin{align}\begin{split}\label{eq:sir}
S(t+1) &= S(t) - \mu(t) \\
I(t+1) &= I(t) + \mu(t) - \gamma I(t) \\
R(t+1) &= R(t) + \gamma I(t)
\end{split}\end{align}

Initial conditions are $S_i(0), I_i(0)$ and $R_i(0) = 1 - S_i(0) -
I_i(0)$. The parameter vector to be estimated is:
\begin{equation*}
\boldsymbol{\phi} = [\theta, S_1(0), I_1(0), S_2(0), I_2(0)]^T \in \mathbb{R}^5
\end{equation*}

\subsection{Influenza}
We focus our analysis on seasonal influenza, which provides extensive
surveillance data for validation. We seek the recovery rate parameter
$\gamma$.

From Keeling \& Rohani~\cite{keeling2008modeling}, the continuous-time
recovery rate for influenza is $\gamma_{\text{cont}} = 1/2.2 \text{
  days}^{-1}$. For weekly discrete-time observations, we must convert
this parameter appropriately.

Consider a scenario where no new infections occur (all susceptibles
removed). After one week:
\begin{align}
\text{Discrete model:} \quad I(t+1) &= (1-\gamma) I(t) \\
\text{Continuous model:} \quad I(t+7) &= \exp(-7\gamma_{\text{cont}}) I(t)
\end{align}

Matching the fraction of infecteds remaining after one week (7 days):
\begin{equation}
1 - \gamma = \exp\left(-\frac{7}{2.2}\right) = \exp(-3.18) \approx 0.041
\end{equation}

Therefore, the calibrated discrete recovery rate is:
\begin{equation}
\gamma = 1 - \exp\left(-\frac{7}{2.2}\right) \approx 0.959
\end{equation}


Here $\gamma$ takes the role of probability of recovery in one
week. Average disease duration is $1/ \gamma$ then (a geometric random
variable). The basic relation of reproduction number is $R_0
\frac{\beta}{\gamma}$. Hence
\begin{equation}
  R_0 = \frac{\beta(t)}{\gamma} = \frac{\beta_0 (1+\epsilon)}{\gamma}
\end{equation}
Taking $\epsilon = 0.3$ and assuming the $R_0$ is cited for peak infectiousness we find
$$
\beta_0 = \frac{R_0 \gamma}{1+\epsilon} = \frac{3.65 (1-\exp\left (-\frac{7}{2.2} \right))}{1+0.3} \approx 2.7
$$

\subsection{Partials wrt ICs}
We will later need to caclulate partials of the state at time $t$ with
respect to initial conditions $y \in \{S_1(0), I_1(0), S_2(0),
I_2(0)\}$, i.e.~$\frac{\partial S(t)}{\partial y}$ and $\frac{\partial
  I(t)}{\partial y}$. We record these below and find them by taking
partials in the original dynamics:

\begin{align}\begin{split}
  \frac{\partial S(t+1)}{\partial y} &= \frac{\partial S(t)}{\partial
  y} - \beta(t) \left[ \frac{\partial S(t)}{\partial y} \circ CI(t) +
  S(t)\circ C \frac{\partial I(t)}{\partial y} \right] \\
  %
  %
  %
  & = \frac{\partial S(t)}{\partial y} - \frac{\partial \mu(t)}{\partial y} \\
  %
  %
  %
  \frac{\partial I(t+1)}{\partial y} &= (1-\gamma) \frac{\partial
  I(t)}{\partial y} + \frac{\partial \mu(t)}{\partial y}.\\
  %
  %
  %
\end{split}\end{align}        
with initial conditions:
\begin{align}\begin{split}
\frac{\partial S_i(0)}{\partial S_j(0)} &= \delta_{ij}, \quad \frac{\partial I_i(0)}{\partial S_j(0)} = 0 \\
%
%
%
\frac{\partial S_i(0)}{\partial I_j(0)} &= 0, \quad \frac{\partial I_i(0)}{\partial I_j(0)} = \delta_{ij}
\end{split}\end{align}



\subsection{The Cram\'er-Rao Lower Bound}
Our goal is to establish fundamental limits on the precision of any
unbiased estimator $\hat{\theta}$ for the connectivity parameter
$\theta$. The \emph{Cram\'er-Rao Lower Bound (CRLB)} provides such
fundamental limits by establishing that for any unbiased estimator:
\begin{equation}
\text{Cov}(\hat{\boldsymbol{\phi}}) \succeq \mathbf{J}^{-1}
\end{equation}
where $\mathbf{J}$ is the \emph{Fisher Information Matrix} (see below)
and $\mathbf{A} \succeq \mathbf{B}$ means $\mathbf{A} - \mathbf{B}$ is
positive semidefinite.

For a currently unspecified observation model with observations
$\mathbf{Y}$ and likelihood $f(\mathbf{Y}|\boldsymbol{\phi})$, the
Fisher Information Matrix can be computed as:

\begin{equation}
  [\mathbf{J}]_{kl} 
  %-\mathbb{E}\left[\frac{\partial^2 \log
   % f(\mathbf{Y}|\boldsymbol{\phi})}{\partial \phi_k \partial
   % \phi_l}\right]
  = \mathbb{E}\left[\frac{\partial \log
      f(\mathbf{Y}|\boldsymbol{\phi})}{\partial \phi_k}\frac{\partial
      \log f(\mathbf{Y}|\boldsymbol{\phi})}{\partial \phi_l}\right]
\end{equation}

We find the CRLB for the connectivity parameter $\theta$ using the

\begin{equation*}
\text{Var}(\hat{\theta}) \geq [\mathbf{J}^{-1}]_{11} = \|L^{-1}\mathbf{e}_1\|^2.
