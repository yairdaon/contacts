we need data pertaining to connectivity: flights, commutes, etc.

\vspace{1em}

{\Large\textbf{Q: Wouldn't it be cool if, instead of relying on
    connectivity data—we could infer connectivity from observations?}}

\vspace{0.5em}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\headerbox{Compartmental Models}{name=models,column=0,below=background}{
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\textbf{Bare bones SIR model}

\vspace{0.3em}

\begin{align*}
\dot{S} &= - \beta(t) \, S \, I \\
\dot{I} &= \beta(t) \, S \, I - \gamma I \\
\dot{R} &= \gamma I
\end{align*}

Seasonal driver: $\boxed{\beta(t) = \beta_0 (1+\delta \sin(t))}$

\vspace{0.8em}

\textbf{Multi-region SEIR with contact matrix}

\vspace{0.3em}

Add an \textbf{Exposed} compartment $\mathbf{E}$ and
demography. $\mathbf{S}, \mathbf{E}, \mathbf{I}, \mathbf{R} \in
\mathbb{R}^n$ with $n = \#\text{regions}$. Connectivity given by
contact matrix $\mathbf{C}$.

\begin{align*}
\dot{\mathbf{S}} &= \mu - \boxed{\beta(t) \, \mathbf{S} \circ \mathbf{C} \mathbf{I}} - \mu \mathbf{S} \\
\dot{\mathbf{E}} &= \boxed{\beta(t) \, \mathbf{S} \circ \mathbf{C} \mathbf{I}} - \sigma \mathbf{E} - \mu \mathbf{E} \\
\dot{\mathbf{I}} &= \sigma \mathbf{E} - \gamma \mathbf{I} - \mu \mathbf{I} \\
\dot{\mathbf{R}} &= \gamma \mathbf{I} - \mu \mathbf{R}
\end{align*}

\vspace{0.3em}

\begin{itemize}
\item $\circ$ is element-wise multiplication.
\item $\mathbf{C}\mathbf{I}$ is matrix product.
\end{itemize}

\vspace{0.3em}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\headerbox{When contact matrix unknown}{name=inverse,column=0,below=models}{
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
We bound performance. Assume $S(t_i), I(t_i)$ for $t_i, i=1,\ldots,T$
known (unrealistic). Observed incidence is:
\begin{align*}
\text{inc}_i &\approx \beta(t_i) \, S(t_i) \circ C I(t_i) \\
\Rightarrow C I(t_i) &\approx \frac{\text{inc}_i}{\beta(t_i) S(t_i)}
\end{align*}

\vspace{1em}

Collect $I(t_i)$ into a matrix:
\begin{equation*}
M = [I(t_1) \mid I(t_2) \mid \cdots \mid I(t_n)]
\end{equation*}

\vspace{1em}

{\Large\textbf{Then:}}
\begin{equation*}
\boxed{{\text{relative error}(C)} \leq {\text{relative error}(\text{inc}_i)} \kappa(M)}
\end{equation*}

where $\kappa(M)$ is the condition number of $M$.
\vspace{0.5em}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\headerbox{Results}{name=results,column=1,below=map}{
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\vspace{0.5em}

Assume noise $\varepsilon_t \sim \mathcal{N}(0, \text{inc}_t)$.

\vspace{1em}
\begin{itemize}
\item Merely 2 regions ($n=2$): $\kappa(M) \approx 10^5$.
\item 1\% measurement error $\Rightarrow$ relative error for $C \approx 10^5 \times 0.01 = 1000$.
\item Inferred off-diagonal of contact matrix $C_{ij} \sim \mathcal{U}(0,1), i\neq j$.
\item Complete loss of information on connectivity.
\end{itemize}

\vspace{1em}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\headerbox{Why Is This So?}{name=hard,column=1,below=results}{
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{align*}
  &\text{seasonal forcing $\beta(t)$} \\
  &\Rightarrow \text{regions synchronize}\\
  &\Rightarrow \text{infection matrix $M$ ill-conditioned}
\end{align*}
}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\headerbox{Conclusion}{name=conclusion,column=1,below=hard}{
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{itemize}
\item \textbf{Inferring contact matrices from epidemic data is extremely ill-conditioned}.
\item Synchrony between regions makes the inverse problem nearly impossible.
\item Traditional connectivity data remains essential.
\end{itemize}
}
