%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\headerbox{Background}{name=background,column=0,below=map}{
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

Sometimes we want to know how infectious diseases spread. To this end,
we need data pertaining to connectivity: flights, commutes, etc.

\vspace{1em}

{\Large\textbf{Q: Wouldn't it be cool if, instead of relying on
    connectivity data—we could infer connectivity from observations?}}

\vspace{0.5em}
}


%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\headerbox{Multi-region SEIR with contact matrix}{name=model,column=0,below=background}{
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\vspace{0.3em}

\begin{itemize}
  \item $\mathbf{S}, \mathbf{E}, \mathbf{I}, \mathbf{R} \in \mathbb{R}^n$ with
    $n = \#\text{regions}$.
  %\item Model:
    \begin{align*}
      \dot{\mathbf{S}} &= \mu - \boxed{\beta(t) \, \mathbf{S} \circ \mathbf{C} \mathbf{I}} - \mu \mathbf{S} \\
      \dot{\mathbf{E}} &= \boxed{\beta(t) \, \mathbf{S} \circ \mathbf{C} \mathbf{I}} - \sigma \mathbf{E} - \mu \mathbf{E} \\
      \dot{\mathbf{I}} &= \sigma \mathbf{E} - \gamma \mathbf{I} - \mu \mathbf{I} \\
      \dot{\mathbf{R}} &= \gamma \mathbf{I} - \mu \mathbf{R}
    \end{align*}
  \item Connectivity given by contact matrix
    $\mathbf{C}, 0\leq c_{ij}\leq 1$.
  \item Seasonal driver: $\beta(t) = \beta_0 (1+\delta \sin(t))$ 
  \item $\gamma^{-1}$ infectiousness period, $\sigma^{-1}$ incubation period. 
  \item $\circ$ element-wise multiplication, $\mathbf{C}\mathbf{I}$ matrix product.
\end{itemize}

\vspace{0.3em}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\headerbox{Inferring the contact matrix}{name=inverse,column=0, below=model}{
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% To bound performance, assume $S(t_i), I(t_i)$ for $t_i, i=1,\ldots,T$
%% known (although unrealistic).
Observed incidence is:
\begin{align*}
\text{inc}_i &\approx \beta(t_i) \, S(t_i) \circ C I(t_i) \\
\Rightarrow C I(t_i) &\approx \frac{\text{inc}_i}{\beta(t_i) S(t_i)}
\end{align*}

\vspace{1em}

Collect $I(t_i)$ into a matrix:
\begin{equation*}
M = \left[
  \begin{array}{cccc}
    \vert & \vert &        & \vert \\
    I(t_1)    & I(t_2)    & \ldots & I(t_T)    \\
    \vert & \vert &        & \vert 
  \end{array}
\right]
\end{equation*}

\vspace{1em}

{\Large\textbf{Then:}}
\begin{equation*}
\boxed{{\text{relative error}(C)} \leq {\text{relative error}(\text{inc}_i)} \kappa(M)}
\end{equation*}

$\kappa(M)$ is the condition number: ratio of maximal and minimal singular values.
%\vspace{0.5em}
}


%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\headerbox{Results}{name=results,column=1, below=map}{
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{itemize}
\item 2 regions: $\kappa(M) \approx 10^5$.
\item For 2 regions, 1\% measurement error $\Rightarrow$ relative error for $C \approx 10^5 \times 0.01 = 1000$.
\item 5 regions: $\kappa(M) \approx 10^8$.
\item 10 regions: $\kappa(M) \approx 10^{11}$.
\item Solving for contact matrix, entries $C_{ij} \sim \mathcal{U}(0,1), i\neq j$.
\item All information on connectivity is lost.
\end{itemize}

\headerbox{Why This Happens?}{name=hard,column=1,below=results}{
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
seasonal forcing $\Rightarrow$ regions synchronize $\Rightarrow$ huge $\kappa(M)$.
}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\headerbox{Conclusion}{name=conclusion,column=1,below=hard}{
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{itemize}
\item \textbf{Inferring contact matrices from synchronized epidemic data is extremely ill-conditioned}.
\item Synchrony between regions makes the inverse problem intractable.
\item Traditional connectivity data remains essential.
\end{itemize}
}
